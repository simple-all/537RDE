\section{Appendix B: Additional Equations}

\begin{equation}
\begin{split}
Re_x=\frac{\rho u x}{\mu}	\\
\text{If } Re_x<500,000, \text{ Laminar boundary layer} \\
\text{If }   Re_x>500,000, \text{ Turbulent boundary layer}
\end{split}
\label{eqn:reynolds}
\end{equation}


\begin{equation}
Pr=\frac{\mu Cp}{K}
\label{eqn:prandtl}
\end{equation}

\begin{equation}
Nu_x=\frac{h_g x}{K}
\label{eqn:nusselt}
\end{equation}
